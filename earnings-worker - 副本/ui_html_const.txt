    <script>
        let chart = null;
        let currentSymbol = '';

        function showStatus(message, type = 'info') {
            const statusDiv = document.getElementById('statusDiv');
            statusDiv.innerHTML = \`<div class="status \\\\$\{type}">\\\\$\{message}</div>\`;
        }
        
        function showLoading() {
            const statusDiv = document.getElementById('statusDiv');
            statusDiv.innerHTML = '<div class="loading"><div class="spinner"></div>Loading data...</div>';
        }
        
        async function updateData() {
            const symbol = document.getElementById('symbolInput').value.toUpperCase().trim();
            if (!symbol) {
                showStatus('Please enter a symbol', 'error');
                return;
            }
            
            showLoading();
            
            try {
                // Update earnings
                const earningsResp = await fetch(\`/api/update?symbol=\\\\$\{symbol}\`, { method: 'POST' });
                const earningsData = await earningsResp.json();
                
                // Update prices
                const pricesResp = await fetch(\`/api/update-prices?symbol=\\\\$\{symbol}\`, { method: 'POST' });
                const pricesData = await pricesResp.json();
                
                if (earningsData.error || pricesData.error) {
                    showStatus(\`Error: \\\\$\{earningsData.error || pricesData.error}\`, 'error');
                } else {
                    showStatus(\`Updated \\\\$\{earningsData.count} earnings records and \\\\$\{pricesData.count} price records\`, 'success');
                    setTimeout(() => analyzeStock(), 2000);
                }
            } catch (e) {
                showStatus(\`Error: \\\\$\{e.message}\`, 'error');
            }
        }
        
        async function analyzeStock() {
            const symbol = document.getElementById('symbolInput').value.toUpperCase().trim();
            if (!symbol) {
                showStatus('Please enter a symbol', 'error');
                return;
            }
            
            currentSymbol = symbol;
            showLoading();
            document.getElementById('resultsSection').classList.add('hidden');
            
            try {
                const response = await fetch(\`/api/forward-peg?symbol=\\\\$\{symbol}\`);
                const data = await response.json();
                
                if (data.error) {
                    showStatus(\`Error: \\\\$\{data.error}\`, 'error');
                    return;
                }
                
                displayResults(data);
                document.getElementById('statusDiv').innerHTML = '';
                document.getElementById('resultsSection').classList.remove('hidden');
                
            } catch (e) {
                showStatus(\`Error: \\\\$\{e.message}\`, 'error');
            }
        }
        
        function displayResults(data) {
            const latest = data.timeSeries[data.timeSeries.length - 1];
            
            // Update header
            document.getElementById('stockSymbol').textContent = data.symbol;
            document.getElementById('stockPrice').textContent = \`$\\\\$\{parseFloat(latest.price).toFixed(2)}\`;
            
            // Update metrics
            document.getElementById('forwardEPS').textContent = \`$\\\\$\{data.currentForwardEPS}\`;
            document.getElementById('forwardPE').textContent = latest.forwardPE.toFixed(2);
            
            const growthEl = document.getElementById('growthRate');
            if (latest.growthRate) {
                growthEl.textContent = \`\\\\$\{latest.growthRate}%\`;
                growthEl.className = 'metric-value ' + (parseFloat(latest.growthRate) > 0 ? 'positive' : 'negative');
            } else {
                growthEl.textContent = 'N/A';
                growthEl.className = 'metric-value';
            }
            
            const pegEl = document.getElementById('pegRatio');
            if (latest.peg) {
                pegEl.textContent = latest.peg;
                pegEl.className = 'metric-value';
            } else {
                pegEl.textContent = 'N/A';
                pegEl.className = 'metric-value';
            }
            
            // Render chart
            renderChart(data.timeSeries);
            
            // Populate table (show last 30 days)
            const tableBody = document.getElementById('dataTableBody');
            tableBody.innerHTML = '';
            
            const recentData = data.timeSeries.slice(-30).reverse();
            recentData.forEach(row => {
                const tr = document.createElement('tr');
                tr.innerHTML = \`
                    <td>\\\\$\{row.date}</td>
                    <td>$\\\\$\{parseFloat(row.price).toFixed(2)}</td>
                    <td>$\\\\$\{parseFloat(row.forwardEPS).toFixed(2)}</td>
                    <td>\\\\$\{row.forwardPE.toFixed(2)}</td>
                    <td>\\\\$\{row.growthRate ? row.growthRate + '%' : 'N/A'}</td>
                    <td>\\\\$\{row.peg || 'N/A'}</td>
                \`;
                tableBody.appendChild(tr);
            });
        }
        
        function renderChart(timeSeries) {
            const ctx = document.getElementById('pegChart').getContext('2d');
            
            if (chart) {
                chart.destroy();
            }
            
            const labels = timeSeries.map(d => d.date);
            const prices = timeSeries.map(d => parseFloat(d.price));
            const pegs = timeSeries.map(d => d.peg ? parseFloat(d.peg) : null);
            
            chart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Price',
                            data: prices,
                            borderColor: '#0078FF',
                            backgroundColor: 'rgba(0, 120, 255, 0.1)',
                            yAxisID: 'y',
                            tension: 0.1,
                            borderWidth: 2
                        },
                        {
                            label: 'PEG Ratio',
                            data: pegs,
                            borderColor: '#FF6B00',
                            backgroundColor: 'rgba(255, 107, 0, 0.1)',
                            yAxisID: 'y1',
                            tension: 0.1,
                            borderWidth: 2
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    interaction: {
                        mode: 'index',
                        intersect: false,
                    },
                    plugins: {
                        legend: {
                            position: 'top',
                            labels: {
                                font: { size: 13, weight: '600' },
                                padding: 15
                            }
                        },
                        tooltip: {
                            backgroundColor: 'rgba(15, 20, 25, 0.95)',
                            padding: 12,
                            titleFont: { size: 13, weight: '600' },
                            bodyFont: { size: 13 }
                        }
                    },
                    scales: {
                        x: {
                            grid: { display: false },
                            ticks: {
                                maxTicksLimit: 10,
                                font: { size: 11 }
                            }
                        },
                        y: {
                            type: 'linear',
                            display: true,
                            position: 'left',
                            title: {
                                display: true,
                                text: 'Price ($)',
                                font: { size: 13, weight: '600' }
                            },
                            grid: { color: '#E1E8ED' }
                        },
                        y1: {
                            type: 'linear',
                            display: true,
                            position: 'right',
                            title: {
                                display: true,
                                text: 'PEG Ratio',
                                font: { size: 13, weight: '600' }
                            },
                            grid: { drawOnChartArea: false }
                        }
                    }
                }
            });
        }
        
        // Auto-analyze on load if symbol is provided

            if (symbol) {
                document.getElementById('symbolInput').value = symbol;
